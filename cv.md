function isIsogram(str) {
  const obj = {};
  const mass = str.toLowerCase().split("");
  for (let i = 0; i <= mass.length - 1; i++) {
    if (!obj.hasOwnProperty(mass[i])) {
      obj[mass[i]] = 0;
    }
    obj[mass[i]] += 1;
  }
  for (key in obj) {
    if (obj[key] > 1) {
      return false;
    }
  }
  return true;
}
